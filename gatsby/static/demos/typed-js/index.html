Installation


# With NPM
npm install typed.js

# With Yarn
yarn add typed.js

# With Bower
bower install typed.js

Setup


var typed = new Typed('.element', {
  strings: ["First sentence.", "Second sentence."],
  typeSpeed: 30
});
